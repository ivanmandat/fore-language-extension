# Аналитическая панель


В примере приведено XML-описание аналитической панели «Аналитическая панель 2». Панель содержит блоки «[Надпись](../Props/block_label.htm)», «[Рисунок](../Props/adhoc_image.htm)», «[HTML](../Props/block_frame.htm)», расположенные на отдельных панелях. Описание однотипных структурных свойств [background](../Props/AdhocSpec_block.htm#background) и [decor](../Props/AdhocSpec_block.htm#decor), а также описание блоков опущено.


<kap version="7.0">

  <block name="Аналитическая панель 2" type="Dashboard" key="CIPJA1WUNLRACX9N">

    <prop tag="defaultBlockStyle">

      <prop tag="blockPanel">

        <prop tag="background"><!-- Фрагмент кода пропущен --></prop>

        <prop tag="decor"><!-- Фрагмент кода пропущен --></prop>

        <prop tag="blockTitle">

          <prop tag="show" val="0" ></prop>

          <prop tag="font">

            <prop tag="isBold" val="1"></prop>

          </prop>

        </prop>

        <prop tag="expandable" val="0"></prop>

      </prop>

      <prop tag="layout">

        <prop tag="anchorTop" val="1"></prop>

        <prop tag="anchorLeft" val="1"></prop>

      </prop>

    </prop>

    <prop tag="name" val="Аналитическая панель 2"></prop>

    <prop tag="background"><!-- Фрагмент кода пропущен --></prop>

    <prop tag="layouts">

      <area key="MU13B5ZP3W8XFEZP">

        <block name="Надпись" type="Label" key="MU13B5ZP3W8XFEZP">

        <!-- Фрагмент кода пропущен -->

        </block>

      </area>

      <area key="PCW1SLGYPYGMRDQ6">

        <block name="Рисунок" type="Image" key="PCW1SLGYPYGMRDQ6">

        <!-- Фрагмент кода пропущен -->

        </block>

      </area>

      <area key="BGJAU9WOSCLZ93JX">

        <block name="HTML" type="Frame" key="BGJAU9WOSCLZ93JX">

        <!-- Фрагмент кода пропущен -->

        </block>

      </area>

    </prop>

  </block>

</kap>
