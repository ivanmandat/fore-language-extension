# Информационные блоки без источника данных


В примерах приведены XML-описания информационных блоков, для которых
 не требуется настройка источника данных. Описание однотипных структурных
 свойств [background](../Props/AdhocSpec_block.htm#background),
 [layout](../Props/AdhocSpec_block.htm#layout),
 [decor](../Props/AdhocSpec_block.htm#decor)
 и [blockTitle](../Props/AdhocSpec_block.htm#blocktitle) опущено.


## Рисунок


<!-- Фрагмент
 кода пропущен -->

<block name="Рисунок"
 type="Image" key="PCW1SLGYPYGMRDQ6">

  <prop tag="name"
 val="Рисунок" />

  <prop tag="background"><!--
 Фрагмент кода пропущен --></prop>

  <prop tag="layout"><!--
 Фрагмент кода пропущен --></prop>

  <prop tag="fillImage"
 val="1" />

  <prop tag="imageData"
 val="data:image/png;base64,iVBORw0KGgo<!--
 Фрагмент кода Base64 пропущен -->lFTkSuQmCC" />

</block>

<!-- Фрагмент кода пропущен -->

 Фрагмент кода пропущен --></prop>

 Фрагмент кода пропущен --></prop>

    <prop tag="show" val="0"
 />

 на повышение эффективности управленческой деятельности."
 />

 кода пропущен --></prop>

    <prop tag="align" val="Center"

  </prop>

</block>
