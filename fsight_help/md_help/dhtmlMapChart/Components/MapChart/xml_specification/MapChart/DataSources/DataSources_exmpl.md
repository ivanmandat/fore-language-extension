# Пример настройки источников данных


На странице приведен пример настройки списка источников данных в формате [JSON](../../../xml_and_json.htm).


Примечание. При автоматической конвертации из формата XML в JSON перед атрибутами появляется знак «@», который можно не удалять.


"DataSources":


 {


   "DataSource":


    [


      {


       "Id": "source",


       "Dimension":


         [


           {"Items" : "35.61 75.38 99.34 85.44","Id" : "RU-YAN"},


           {"Items" : "0.71 89.36 3.43 75.68","Id" : "RU-KHM"},


           {"Items" : "67.71 93.96 90.01 26.17","Id" : "RU-TYU"},


           {"Items" : "50.59 35.48 96.02 98.28","Id" : "RU-NEN"}


         ]


      },


      {


       "Id" : "BarBackSource",


       "Dimension" :


         [


           {"Items" : "1 2 3 4","Id" : "RU-YAN"},


           {"Items" : "1 2 3 4","Id" : "RU-KHM"},


           {"Items" : "1 2 3 4","Id" : "RU-TYU"},


           {"Items" : "1 2 3 4","Id" : "RU-NEN"}


         ]


       }


    ]


}
